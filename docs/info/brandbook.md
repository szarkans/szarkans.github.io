<script>
  function copyToClipboard(element) {
    const text = element.getAttribute('data-clipboard-text');
    navigator.clipboard.writeText(text).then(() => {
      console.log('Скопирован цвет: ' + text); 
    }, (err) => {
      console.error('Ошибка копирования цвета: ', err);
    });
  }
</script>
